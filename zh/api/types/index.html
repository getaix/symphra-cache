    class EvictionPolicy(str, Enum):
    """
    缓存淘汰策略枚举

    定义缓存满时的淘汰规则：
    - LRU: Least Recently Used（最近最少使用），淘汰最久未访问的项
    - LFU: Least Frequently Used（最不经常使用），淘汰访问次数最少的项
    - FIFO: First In First Out（先进先出），淘汰最早插入的项
    """

    LRU = "lru"  # 最近最少使用
    LFU = "lfu"  # 最不经常使用
    FIFO = "fifo"  # 先进先出
